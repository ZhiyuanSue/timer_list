6use alloc::{boxed::Box, collections::BinaryHeap};
7use core::cmp::{Ord, Ordering, PartialOrd};
8use core::time::Duration;

10/// The type of the time value.
11///
12/// Currently it is just an alias of [`core::time::Duration`].
13pub type TimeValue = Duration;
14
15/// A trait that all timed events must implement.
16pub trait TimerEvent {
17    /// Callback function that will be called when the timer expires.
18    fn callback(self, now: TimeValue);
19}
20
21struct TimerEventWrapper<E> {
22    deadline: TimeValue,
23    event: E,
24}
25
26/// A list of timed events.
27///
28/// It internally uses a min-heap to store the events by deadline, make it
29/// possible to trigger these events sequentially.
30pub struct TimerList<E: TimerEvent> {
31    events: BinaryHeap<TimerEventWrapper<E>>,
32}
33
34impl<E> PartialOrd for TimerEventWrapper<E> {
35    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
36        Some(self.cmp(other)) // reverse ordering for Min-heap
37    }
38}
39
40impl<E> Ord for TimerEventWrapper<E> {
41    fn cmp(&self, other: &Self) -> Ordering {
42        other.deadline.cmp(&self.deadline) // reverse ordering for Min-heap
43    }
44}
45
46impl<E> PartialEq for TimerEventWrapper<E> {
47    fn eq(&self, other: &Self) -> bool {
48        self.deadline == other.deadline
49    }
50}
51
52impl<E> Eq for TimerEventWrapper<E> {}
53
54impl<E: TimerEvent> TimerList<E> {
55    /// Creates a new empty timer list.
56    pub fn new() -> Self {
57        Self {
58            events: BinaryHeap::new(),
59        }
60    }
61
62    /// Whether there is no timed event.
63    #[inline]
64    pub fn is_empty(&self) -> bool {
65        self.events.is_empty()
66    }
67
68    /// Set a timed event that will be triggered at `deadline`.
69    pub fn set(&mut self, deadline: TimeValue, event: E) {
70        self.events.push(TimerEventWrapper { deadline, event });
71    }
72
73    /// Cancel all events that meet the condition.
74    pub fn cancel<F>(&mut self, condition: F)
75    where
76        F: Fn(&E) -> bool,
77    {
78        // TODO: performance optimization
79        self.events.retain(|e| !condition(&e.event));
80    }
81
82    /// Get the deadline of the most recent event.
83    #[inline]
84    pub fn next_deadline(&self) -> Option<TimeValue> {
85        self.events.peek().map(|e| e.deadline)
86    }
87
88    /// Try to expire the earliest event that passed the deadline at the given
89    /// time.
90    ///
91    /// Returns `None` if no event is expired.
92    pub fn expire_one(&mut self, now: TimeValue) -> Option<(TimeValue, E)> {
93        if let Some(e) = self.events.peek() {
94            if e.deadline <= now {
95                return self.events.pop().map(|e| (e.deadline, e.event));
96            }
97        }
98        None
99    }
100}
101
102impl<E: TimerEvent> Default for TimerList<E> {
103    fn default() -> Self {
104        Self::new()
105    }
106}

108/// A simple wrapper of a closure that implements the [`TimerEvent`] trait.
109///
110/// So that it can be used as in the [`TimerList`].
111pub struct TimerEventFn(Box<dyn FnOnce(TimeValue) + 'static>);
112
113impl TimerEventFn {
114    /// Constructs a new [`TimerEventFn`] from a closure.
115    pub fn new<F>(f: F) -> Self
116    where
117        F: FnOnce(TimeValue) + 'static,
118    {
119        Self(Box::new(f))
120    }
121}
122
123impl TimerEvent for TimerEventFn {
124    fn callback(self, now: TimeValue) {
125        (self.0)(now)
126    }
127}

130mod tests {
131    use super::{TimeValue, TimerEvent, TimerEventFn, TimerList};
132    use core::sync::atomic::{AtomicUsize, Ordering};
133    use std::time::{Duration, Instant};
134
135    #[test]
136    fn test_timer_list() {
137        const EVENT_ORDER: &[usize; 4] = &[1, 4, 3, 0]; // timer 2 was canceled
138        static COUNT: AtomicUsize = AtomicUsize::new(0);
139
140        struct TestTimerEvent(usize, TimeValue);
141
142        impl TimerEvent for TestTimerEvent {
143            fn callback(self, now: TimeValue) {
144                let idx = COUNT.fetch_add(1, Ordering::SeqCst);
145                assert_eq!(self.0, EVENT_ORDER[idx]);
146                println!(
147                    "timer {} expired at {:?}, delta = {:?}",
148                    self.0,
149                    now,
150                    now - self.1
151                );
152            }
153        }
154
155        let mut timer_list = TimerList::new();
156        let start_time = Instant::now();
157        let deadlines = [
158            Duration::new(3, 0),            // 3.0 sec
159            Duration::from_micros(500_000), // 0.5 sec
160            Duration::from_secs(4),         // 4.0 sec, canceled
161            Duration::new(2, 990_000_000),  // 2.99 sec
162            Duration::from_millis(1000),    // 1.0 sec,
163        ];
164
165        for (i, &ddl) in deadlines.iter().enumerate() {
166            timer_list.set(ddl, TestTimerEvent(i, ddl));
167        }
168
169        while !timer_list.is_empty() {
170            let now = Instant::now().duration_since(start_time);
171            if now.as_secs() > 3 {
172                timer_list.cancel(|e| e.0 == 2);
173            }
174            while let Some((_deadline, event)) = timer_list.expire_one(now) {
175                event.callback(now);
176            }
177            std::thread::sleep(Duration::from_millis(10)); // sleep 10 ms
178        }
179
180        assert_eq!(COUNT.load(Ordering::SeqCst), 4);
181    }
